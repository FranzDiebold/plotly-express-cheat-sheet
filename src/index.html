Plotly Express cheat sheet

General

Install

pip install plotly

Import

import plotly.express as px

General usage

fig = px.chart_type(
    df,
    **chart_specific_parameters,
    title="Chart title",
    labels={"x_column_name": "X column name"},
    width=600,
    height=400,
)
fig.show()

Chart types

Scatter

px.scatter(
    df,
    x="sepal_width",
    y="sepal_length",
    color="species",
    marginal_y="violin",
    marginal_x="box",
    trendline="ols",
    hover_data=["petal_length", "petal_width"],
)
by Franz Diebold (diebold.io)

Scatter matrix

px.scatter_matrix(
    df,
    dimensions=[
      "sepal_width", "sepal_length",
      "petal_width", "petal_length"
    ],
    color="species",
)

Histogram

px.histogram(
    df,
    x="sepal_width",
    color="species",
    marginal="box",  # or "rug", "violin"
    nbins=30,
)

Box

px.box(
    df,
    x="species", y="sepal_width",
    notched=True, points="all"
)

Density heatmap (2D Histogram)

px.density_heatmap(
    df,
    x="sepal_width",
    y="sepal_length",
    nbinsx=20,
    nbinsy=20,
    histfunc="count",  # or "sum", "avg"
    marginal_x="rug",
    marginal_y="histogram",
)

Bar

px.bar(count_df, x="species", y="count")

Pie

px.pie(count_df,
       names="species", values="count")

More options

Facet plots

px.density_heatmap(
    df,
    x="sepal_width",
    y="sepal_length",
    facet_row="species",
